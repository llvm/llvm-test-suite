# selected nor -ffast-math is passed.

if (NOT TEST_SUITE_BENCHMARKING_ONLY AND
    NOT ${COMBINED_CXX_FLAGS} MATCHES ".*-ffast-math.*")
  list(APPEND CPPFLAGS -DBENCH_AND_VERIFY)
endif()
